namespace PolarSharp.MultiTenant
{
    public interface IFakeDataAware
    {
        bool IsFakeData { get; }
    }
    public interface IMultiTenantPolarClientFactory
    {
        PolarSharp.PolarClient GetClientForCurrentTenant();
    }
    public interface ITenantOwned : PolarSharp.MultiTenant.IFakeDataAware
    {
        string TenantId { get; }
    }
    public class PolarMultiTenantOptions
    {
        public PolarMultiTenantOptions() { }
        public PolarSharp.MultiTenant.StrategyOptions.ClaimStrategyOptions Claim { get; set; }
        public PolarSharp.MultiTenant.StrategyOptions.HeaderStrategyOptions Header { get; set; }
        public PolarSharp.MultiTenant.StrategyOptions.HostnameStrategyOptions Hostname { get; set; }
        public PolarSharp.MultiTenant.StrategyOptions.RouteStrategyOptions Route { get; set; }
        public PolarSharp.MultiTenant.TenantStrategy Strategy { get; set; }
        public System.Collections.Generic.List<PolarSharp.MultiTenant.PolarTenantInfo> Tenants { get; set; }
    }
    public sealed class PolarTenantInfo : Finbuckle.MultiTenant.Abstractions.ITenantInfo
    {
        public PolarTenantInfo() { }
        public string Id { get; set; }
        public string Identifier { get; set; }
        public string? Name { get; set; }
        public string PolarAccessToken { get; set; }
        public string PolarOrganizationId { get; set; }
        public PolarSharp.PolarServer Server { get; set; }
        public System.Guid TenantId { get; }
        public string? WebhookSecret { get; set; }
    }
    public enum TenantStrategy
    {
        Header = 0,
        Route = 1,
        Hostname = 2,
        Claim = 3,
    }
}
